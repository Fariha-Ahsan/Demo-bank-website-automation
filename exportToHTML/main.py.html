import time
import unittest
from selenium import webdriver
from selenium.webdriver import ActionChains
from selenium.webdriver.chrome.service import Service
from selenium.webdriver.common.keys import Keys
from selenium.webdriver.common.by import By
from webdriver_manager.chrome import ChromeDriverManager


class TestDarazAutomation(unittest.TestCase):

    def setUp(self):
        self.driver = webdriver.Chrome(service=Service(ChromeDriverManager().install()))
        self.base_url="https://www.daraz.com.bd/?spm=a2a0e.home.header.dhome.26a712f7FqmeUv"
        self.driver.implicitly_wait(30)


    def test_category(self):
        driver=self.driver
        driver.get(self.base_url)
        cat_btn=driver.find_element(By.XPATH,"//span[@class='lzd-site-menu-nav-category-text']")
        actionchain=ActionChains(driver)
        actionchain.move_to_element(cat_btn).perform()
        time.sleep(5)
        driver.find_element(By.XPATH,"//span[normalize-space()='Health & Beauty']").click()
        time.sleep(5)
    def test_search_in_homepage(self):
        driver = self.driver
        elem = driver.find_element(By.ID, "q")
        elem.send_keys("mobile")
        elem.send_keys(Keys.RETURN)
        #get_url=driver.current_url
        #print("new url is"+str(get_url))
        self.assertNotIn("No results found.", driver.page_source)





    def tearDown(self):

        self.driver.close()


if __name__ == "__main__":
    unittest.main()
